def verificar_vitoria(tabuleiro, simbolo):

            return True
    if all(tabuleiro[i][i] == simbolo for i in range(3)):
        return True
    if all(tabuleiro[i][2-i] == simbolo for i in range(3)):
        return True
    return False

    # 1. Tentar vencer
